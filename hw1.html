Codice Utilizzato

HTML
<div id="box">
    <p id="testo-dinamico">In attesa di interazione...</p>
</div>

<div class="controls">
    <button onclick="cambiaContenuto()">Cambia Testo</button>
    <button onclick="cambiaColore()">Cambia Stile</button>
    <button onclick="aggiungiElemento()">Aggiungi Nodo</button>
    <button onclick="resetBox()">Reset</button>
</div>
JavaScript
function cambiaContenuto() {
    const testo = document.getElementById("testo-dinamico");
    testo.innerHTML = "Contenuto aggiornato via JS!";
}

function cambiaColore() {
    const box = document.getElementById("box");
    box.style.backgroundColor = "#e3f2fd";
    box.style.border = "2px solid #1a1a1a";
}

function aggiungiElemento() {
    const box = document.getElementById("box");
    const nuovo = document.createElement("p");
    nuovo.innerText = "Nuovo elemento aggiunto!";
    box.appendChild(nuovo);
}

function resetBox() {
    location.reload();
}
